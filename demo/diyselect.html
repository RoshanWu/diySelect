jquery-diyselect

A simple jQuery plugin used to diy the default select in HTML

common use

$("#example1").diySelect();

custom placeholder text of select

$("#example2").diySelect({checkText: 'please choose me'});

select resize

$("#example3").diySelect({autoWidth: true});

custom select width

$("#example4").diySelect({width: 200});

max limited options to display

$("#example5").diySelect({maxLength: 5});

callback when choosing or selecting

$("#example6").diySelect({
  callBackSelect: function () {alert('starting select...')},
  callBackChoose: function () {alert('you choose the "' + $(this).text() + '"')}
});

revert select

$("#example7Button").click(function () {$("#example7").revertSelect();})

choose event trigger

$("#example8Button1").click(function () {
  $("#example8").chooseSelect({val: "item2"});
});
$("#example8Button2").click(function () {
  $("#example8").chooseSelect({index: 3});
});

ajax support

$("#prov").diySelect({
  ajaxUrl: '../demo/region.json',
  parseData: function (data) {
    var terms = [],
        arrStr = data['provinces'].split(','),
        arrCode = data['provcodes'].split(',');

    $.each(arrStr, function (key) {
      terms.push('<li data-option-val="' +
          arrCode[key] + '">' +
          arrStr[key] + '</li>'
      );
    });

    $(this).html(terms.join(''));
  }
});
